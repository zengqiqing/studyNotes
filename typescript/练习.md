- 获取数组长度

  ```typescript
  let food:number[] = []
  food = [1,2,3]
  let len1 = food.length
  console.log(len1) //ok
  
  let coco:string
  coco = "1111"
  let len2 = coco.length//ok
  
  let kevin:boolean
  let len3 = kevin.length//error,布尔值没有length属性
  ```

  

- 创建一个对象

  ```typescript
  interface Person {
      name:string,
      age:number
  }
  let person = {
      name:'coco',
      age:11
  }
   console.log(person,'-- person --')
  
  //为对象新增一个新的属性
  interface Person {
      name:string,
      age:number,
      [x:string]:any
  }
  
  let p1 = {}
  let p2 = p1 as Person
  p2.sex = 'ddd'
  ```
